4 Firestore security rules (basic)

Place in firestore.rules:

rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{userId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == userId;
      allow update: if request.auth != null && request.auth.uid == userId;
    }

    match /donations/{donationId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.resource.data.donorId == request.auth.uid;
      allow update: if request.auth != null && (
          request.auth.uid == resource.data.donorId  // donor can update own donation
          || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin'
          || (request.auth.uid == resource.data.claimedBy) // NGO who claimed can update status
      );
      allow delete: if false; // use admin action to remove
    }

    // add rules for storage and notifications similarly
  }
}

import { createUserWithEmailAndPassword } from "https://www.gstatic.com/firebasejs/9.22.0/firebase-auth.js";
import { doc, setDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/9.22.0/firebase-firestore.js";
import { auth, db } from './firebase-init.js';

async function register(email, password, name, role) {
  const cred = await createUserWithEmailAndPassword(auth, email, password);
  const uid = cred.user.uid;
  await setDoc(doc(db, 'users', uid), {
    name, email, role,
    createdAt: serverTimestamp()
  });
  // redirect to role-specific dashboard
}

6 Donation CRUD + image upload

Implement:

  • Donation creation: upload image to Storage get URL create donations doc with metadata.

  • Claiming: NGO sets status = "claimed" and claimedBy = ngoId. Use transaction to avoid race conditions.

  • Mark picked-up: donor or admin updates status="picked_up".
    Deliverable: donate.html + public/js/donations.js with create/read/update functions.

Transaction example (claim):

js
import { doc, runTransaction } from "https://www.gstatic.com/firebasejs/9.22.0/firebase-firestore.js";

async function claimDonation(donationId, ngoId) {
  const donationRef = doc(db, 'donations', donationId);
  await runTransaction(db, async (tx) => {
    const snap = await tx.get(donationRef);
    if (!snap.exists()) throw "Donation missing";
    const status = snap.data().status;
    if (status !== 'available') throw "Already claimed";
    tx.update(donationRef, { status: 'claimed', claimedBy: ngoId });
  });
}
